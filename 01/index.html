011関数(function)

012アロー関数

(引数) => {処理内容}    //関数を定義

//従来の関数式
var fn = function () {};

//アロー関数
var fn = () => {};

//「関数宣言」をアロー関数に置き換えることは出来ない
function fn(x) {}
  • 関数を簡略化して記述できる
  • thisを束縛できる
  • 引数が1つの場合括弧を省略できる
  • アロー関数内の処理が1行のときは、{}とreturnを省略できる

引数の省略

//引数なし
const myFunc1 = () => {};

//引数1つ
const myFunc2 = a => {};

//引数2つ以上
const myFunc3 = (a,b,c) => {};

{}とreturnの省略

//aを受け取って、a+2を返す関数

const myFunc = (a) => {
return a + 2;
};

const myFunc = (a) => a + 2;    //省略できる
https://analogic.jp/arrow-function/

013関数に渡す値の初期値

function 関数名(引数 = 初期値) {}
  • 関数の引数では「値=初期値」で初期値を定義できる
function calcFunc (price, tax = 0.08) {
  const result = price + price * tax;
  return result;
};

//taxの引数を省略(初期値のまま)
const result1 = calcFunc(100);    //結果108

//taxの引数を指定
const result2 = calcFunc(100, 0.1);    //結果110

014関数に任意の数の引数を渡す(残余引数)

function 関数名(...引数) {}    //渡した引数が配列になる
引数[インデックス]    //インデックスを指定して引数を使用
  • 残余引数は「...引数」で不特定数の引数を受け取れる

引数の合計値を返す関数


function calcFunc (...prices) {
  let result = 0;
  for (const value of prices) {
    result += value;    //受け取った引数すべてを加算
  }
  return result;
};

constresult = calcFunc(10,20,30);    //結果60

015条件に応じて処理を分ける（if文）

if (条件1) {処理1}    //条件1を満たせば処理1を行う
else if (条件2) {処理2}    //条件1を満たさず、条件2を満たせば処理2を行う
else {処理3}    //条件1も条件2も満たさなければ、処理3を行う

alert出し分けの例

const num = 100;

if(num >= 50) {
  alert('50以上です');    //満たすので実行
} else if (num >= 10) {
  alert('10以上50未満です');
} else {
  alert('10未満です');
}

016条件に応じて処理を分ける（switch文）

switch(判定式)    //判定式に応じて処理を分ける
case 値: 処理    //条件を満たせば、実行
default: 処理    //どの条件も満たさない場合、実行（省略可）
break    //処理の終了を明示

alert出し分けの例

const myFruits = 'りんご';

switch(myFruits) {
  case 'りんご':
    alert('りんごです');    //満たすので実行
    break;    //記載しないと次の処理に進んでしまう
  case 'みかん':
    alert('みかんです');
    break;
  default:
    alert('その他です');
    break;
}

switchの式は厳密な等価(===)

const num = '100';

switch(num) {
  case 100:
    alert('true');    //文字列の100ではないので実行されない
    break;
  default:
    alert('false');    //実行される
    break;
}

017処理を繰り返す（for文）

for(初期化処理; 反復の条件; 反復終了の処理) {反復処理}

SAMPLE #017数字を順番に出力する

for (let i = 0; i < 10; i++) {
	console.log(i);
}

SAMPLE #017

  • 1.「初期化処理」が実行、iに0が代入
  • 2.「反復の条件」であるi < 10を満たし、console.log(i);が実行（0の出力）
  • 3.「反復の終わり」でi++が実行、iが1になる
  • 4.「反復の条件」であるi < 10を満たし、console.log(i);が実行（1の出力）
  • 5.「反復の終わり」でi++が実行、iが2になる
  • 6.（・・・繰り返し・・・）
  • 7.「反復の終わり」でi++が実行、iが10になる
  • 8.「反復の条件」iが10未満を満たさないので、処理が終了

018処理を繰り返す（while文）

while(反復の条件) {反復処理}
  • whileは、条件を満たす限り処理を繰り返す
  • forと違い、反復条件しか記載しない

SAMPLE #018数字を順番に出力する

let num = 0;

while(num < 10) {
  console.log(num);
  num += 1;
}

019反復処理をスキップ（continue）

  • 反復処理を抜けたい場合、continueを使うと以降の処理がスキップされ、次のループに移行する。

SAMPLE #019処理スキップの例

for (let i = 0; i < 10; i++) {
  if (i % 2 === 0) {    //iが偶数の場合スキップ
    continue;
  }
  console.log(i);    //奇数の場合、実行
}
